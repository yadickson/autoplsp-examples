```sql
CREATE OR REPLACE PROCEDURE SP_TEST_CURSOR
(
    PO_CURSOR          OUT  SYS_REFCURSOR,
    PO_COD_RETORNO     OUT  NUMBER,
    PO_MSG_RETORNO     OUT  VARCHAR2
)
AS
BEGIN

    PO_COD_RETORNO := 0;
    PO_MSG_RETORNO := 'OK';

    OPEN PO_CURSOR FOR
    SELECT
        0 AS C_NUMERIC,
        0 AS C_INTEGER,
        0 AS C_INT,
        0 AS C_SMALLINTEGER,
        0 AS C_DECIMAL,
        0 AS C_DEC,
        0 AS C_FLOAT,
        0 AS C_REAL,
        ' ' AS C_CHAR,
        TO_CLOB(' ') AS C_CLOB,
        TO_CLOB(' ') AS C_NCLOB,
        TO_BLOB(0) AS C_BLOB
    FROM DUAL;

EXCEPTION
   WHEN OTHERS
   THEN
        PO_COD_RETORNO := 10;
        PO_MSG_RETORNO := 'NOK';

END;
```
